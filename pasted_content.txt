    image = Image.open(ICON_PNG)
    def on_quit(icon_obj, item):

        janela.quit()
        sys.exit()
    def on_show(icon_obj, item):
        janela.after(0, janela.deiconify)
        icon_obj.visible = False
    icon = Icon(
        'Pousadas',
        image,
        menu=Menu(
            MenuItem('Mostrar Painel', on_show),
            MenuItem('Sair', on_quit)
        )
    )
    icon.run()
